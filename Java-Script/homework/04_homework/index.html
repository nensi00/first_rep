JavaScript Functions Examples

1. Різні варіанти створення функцій

// Function Declaration
function hello() { console.log("Hello"); }

// Function Expression
const helloExp = function() { console.log("Hello"); };

// Arrow Function
const helloArrow = () => { console.log("Hello"); };

// Function Constructor
const helloConstructor = new Function('console.log("Hello");');

2. Функція для підрахунку аргументів

function countArguments() { return arguments.length; }

// Arrow Function з використанням rest-параметрів
const countArgumentsArrow = (...args) => args.length;

3. Функція для порівняння двох чисел

function compareNumbers(a, b) {
  if (a < b) return -1;
  if (a > b) return 1;
  return 0;
}

4. Функція обчислення факторіалу

function factorial(n) {
  if (n < 0) return undefined;
  if (n === 0 || n === 1) return 1;
  return n * factorial(n - 1);
}

5. Функція для об'єднання трьох цифр в одне число

function combineDigits(a, b, c) {
  return Number('' + a + b + c);
}

6. Функція обчислення площі

function calcArea(length, width) {
  if (width === undefined) width = length;
  return length * width;
}
